#3)Armstrong or not

num = int(input("Enter a number: "))  # Get user input
t = num
a = 0

while t > 0:
    r = t % 10
    t = t // 10
    a = a + (r ** len(str(num)))

if a == num:
    print("Armstrong number")
else:
    print("Not an Armstrong number")
