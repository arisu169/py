#3)Armstrong or not

num = int(input("Enter a number: "))  # Get user input
t = num
a = 0

while t > 0:
    r = t % 10  # Extract last digit
    t = t // 10  # Remove last digit
    a = a + (r ** len(str(num)))  # Add digit raised to power of total digits

if a == num:
    print("Armstrong number")
else:
    print("Not an Armstrong number")
