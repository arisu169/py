# 14. Write a Python Program to Demonstrate Basic Operations on a Single NumPy Array.
import numpy as np  

arr = np.array([3, 4, 5, 6, 7, 87])  
print(arr)  

print(np.sum(arr))  
print(np.mean(arr))  
print(np.max(arr))  
print(np.min(arr))  
print(arr.size)  

print(arr + 4)  
print(arr - 8)  
print(arr * 3)  
print(arr / 2)
